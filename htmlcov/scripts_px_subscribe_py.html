122 async def session_dedup_loop(self, config, secret, ws_url, topic, pubsub_node_name): 

123 ''' 

124 Subscription loop specifically for ISE pxGrid sessionTopic events. The 

125 logic for de-duplication is based around callingStationId, timestamp and 

126 event content. Multiple events may have the same callimgStationId and 

127 timestamp, but attribute changes, like profiling determining the operating 

128 system for a device, may result in events that have the same timestamp but 

129 different contents. 

130 

131 The algorithm in this routine takes this into account, and will "de- 

132 duplicate" the events (i.e. tell you when a duplicate event arrived). It 

133 uses MD5 (for speed) on a key-sorted dump of the event (which ensures that 

134 duplicate events are detected by the hash digest differing.) 

135 ''' 

136 self.logger.debug('starting subscription to %s at %s', topic, ws_url) 

137 assert topic == '/topic/com.cisco.ise.session', '%s is not the sessionTopic' 

138 

139 ws = WebSocketStomp(ws_url, config.node_name, secret, config.ssl_context) 

140 await ws.connect() 

141 await ws.stomp_connect(pubsub_node_name) 

142 for topic_item in topic: 

143 await ws.stomp_subscribe(topic_item) 

144 try: 

145 while True: 

146 if self.stopped(): 

147 break 

148 message = json.loads(await ws.stomp_read_message()) 

149 with self.dedup_lock: 

150 for s in message['sessions']: 

151 event_text = json.dumps(s, indent=2, sort_keys=True) 

152 event_hash = hashlib.md5(event_text.encode()).hexdigest() 

153 event_key = '{}:{}:{}'.format( 

154 s['callingStationId'], s['timestamp'], event_hash) 

155 if self.event_keys.get(event_key): 

156 self.event_keys[event_key]['count'] = self.event_keys[event_key]['count'] + 1 

157 print('duplicate mac:timestamp:hash event, count {}'.format( 

158 self.event_keys[event_key]['count'])) 

159 print(' --> {}'.format(ws_url)) 

160 else: 

161 self.event_keys[event_key] = {} 

162 self.event_keys[event_key]['count'] = 1 

163 self.event_keys[event_key]['time'] = time.time() 

164 self.event_keys[event_key]['event'] = event_text 

165 self.event_keys[event_key]['md5'] = event_hash 

166 print('{}\nevent from {}'.format('-' * 75, ws_url)) 

167 print(json.dumps(s, indent=2, sort_keys=True)) 

168 sys.stdout.flush() 

169 except asyncio.CancelledError: 

170 pass 

171 self.logger.debug('shutting down listener...') 

172 await ws.stomp_disconnect('123') 

173 await asyncio.sleep(2.0) 

174 await ws.disconnect()
