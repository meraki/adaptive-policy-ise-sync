266def sync_dashboard(): 

267 log = [] 

268 msg = "SYNC_DASHBOARD-NO_ACTION_REQUIRED" 

269 append_log(log, "dashboard_monitor::sync_dashboard::Checking Dashboard Accounts for re-sync...") 

270 

271 # If we know that something needs to be created, do that first. 

272 sss = SyncSession.objects.all() 

273 for ss in sss: 

274 digest_database_data(ss, log) 

275 ss.dashboard.force_rebuild = True 

276 ss.save() 

277 msg = "SYNC_DASHBOARD-CHANGES_MADE_FORCE_UPDATE" 

278 

279 # Ensure that ISE has already completed a sync if it is the source of truth 

280 stat = SyncSession.objects.filter(Q(ise_source=False) | 

281 (Q(iseserver__last_sync__isnull=False) & 

282 Q(dashboard__last_sync__isnull=True)) | 

283 (Q(iseserver__last_sync__isnull=False) & 

284 Q(iseserver__last_sync__gte=F('dashboard__last_sync')))) 

285 if len(stat) <= 0: 

286 append_log(log, "dashboard_monitor::sync_dashboard::Skipping sync as ISE is primary and needs to sync first.") 

287 msg = "SYNC_DASHBOARD-ISE_NEEDS_SYNC" 

288 else: 

289 append_log(log, "dashboard_monitor::sync_dashboard::Running sync") 

290 

291 for s in stat: 

292 ctime = make_aware(datetime.datetime.now()) - datetime.timedelta(seconds=s.sync_interval) 

293 # Perform sync if one of the following conditions is met 

294 # 1) The Sync Session is set to Force Rebuild (this one shouldn't be seen here. but just in case...) 

295 # 2) The Dashboard Instance is set to Force Rebuild 

296 # 3) The timestamp of the Dashboard database object isn't the same as the timestamp of it's last sync 

297 # 4) The timestamp of the Dashboard database object's last sync is beyond the configured manual sync timer 

298 dbs = SyncSession.objects.filter(Q(dashboard__force_rebuild=True) | 

299 Q(force_rebuild=True) | 

300 ~Q(dashboard__last_sync=F('dashboard__last_update')) | 

301 Q(dashboard__last_sync__lte=ctime)) 

302 for d in dbs: 

303 # Log the reason(s) for the current sync 

304 if d.force_rebuild: # pragma: no cover 

305 append_log(log, "dashboard_monitor::sync_dashboard::Sync Session Force Rebuild", d) 

306 msg = "SYNC_DASHBOARD-SYNCSESSION_FORCE_REBUILD" 

307 d.force_rebuild = False 

308 d.save() 

309 if d.dashboard.force_rebuild: 

310 append_log(log, "dashboard_monitor::sync_dashboard::Dashboard Force Rebuild", d) 

311 msg = "SYNC_DASHBOARD-DASHBOARD_FORCE_REBUILD" 

312 d.dashboard.force_rebuild = False 

313 d.dashboard.save() 

314 if d.dashboard.last_sync != d.dashboard.last_update: 

315 append_log(log, "dashboard_monitor::sync_dashboard::Database Config / Sync Timestamp Mismatch", d) 

316 msg = "SYNC_DASHBOARD-CONFIG_SYNC_TIMESTAMP_MISMATCH" 

317 if d.dashboard.last_sync and (d.dashboard.last_sync <= ctime): 

318 append_log(log, "dashboard_monitor::sync_dashboard::Past Manual Sync Interval", d) 

319 msg = "SYNC_DASHBOARD-PAST_SYNC_INTERVAL" 

320 

321 ingest_dashboard_data(dbs, log) 

322 

323 # After ingesting data, more updates may be required (Should these have been caught elsewhere?) 

324 sss = SyncSession.objects.all() 

325 for ss in sss: 

326 digest_database_data(ss, log) 

327 

328 append_log(log, "dashboard_monitor::sync_dashboard::Done") 

329 db_log("dashboard_monitor", log) 

330 return msg, log
