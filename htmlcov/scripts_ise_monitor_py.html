304def sync_ise(): 

305 log = [] 

306 msg = "SYNC_ISE-NO_ACTION_REQUIRED" 

307 append_log(log, "ise_monitor::sync_ise::Checking ISE Accounts for re-sync...") 

308 

309 # If we know that something needs to be created, do that first. 

310 sss = SyncSession.objects.all() 

311 for ss in sss: 

312 digest_database_data(ss, log) 

313 ss.iseserver.force_rebuild = True 

314 ss.iseserver.skip_update = True 

315 ss.iseserver.save() 

316 ss.save() 

317 msg = "SYNC_ISE-CHANGES_MADE_FORCE_UPDATE" 

318 

319 # Ensure that Meraki Dashboard has already completed a sync if it is the source of truth 

320 stat = SyncSession.objects.filter(Q(ise_source=True) | 

321 (Q(dashboard__last_sync__isnull=False) & 

322 Q(iseserver__last_sync__isnull=True)) | 

323 (Q(dashboard__last_sync__isnull=False) & 

324 Q(dashboard__last_sync__gte=F('iseserver__last_sync')))) 

325 

326 if len(stat) <= 0: 

327 append_log(log, "ise_monitor::sync_ise::Skipping sync as Meraki is primary and hasn't synced.") 

328 msg = "SYNC_ISE-DASHBOARD_NEEDS_SYNC" 

329 else: 

330 append_log(log, "ise_monitor::sync_ise::Running sync") 

331 

332 for s in stat: 

333 ctime = make_aware(datetime.datetime.now()) - datetime.timedelta(seconds=s.sync_interval) 

334 # Perform sync if one of the following conditions is met 

335 # 1) The Sync Session is set to Force Rebuild (this one shouldn't be seen here. but just in case...) 

336 # 2) The Dashboard Instance is set to Force Rebuild 

337 # 3) The timestamp of the ISE Server database object isn't the same as the timestamp of it's last sync 

338 # 4) The timestamp of the ISE Server database object's last sync is beyond the configured manual sync timer 

339 dbs = SyncSession.objects.filter(Q(iseserver__force_rebuild=True) | 

340 Q(force_rebuild=True) | 

341 ~Q(iseserver__last_sync=F('iseserver__last_update')) | 

342 Q(iseserver__last_sync__lte=ctime)) 

343 for d in dbs: 

344 # Log the reason(s) for the current sync 

345 if d.force_rebuild: # pragma: no cover 

346 append_log(log, "ise_monitor::sync_ise::Sync Session Force Rebuild", d) 

347 msg = "SYNC_ISE-SYNCSESSION_FORCE_REBUILD" 

348 d.force_rebuild = False 

349 d.save() 

350 if d.iseserver.force_rebuild: 

351 append_log(log, "ise_monitor::sync_ise::Dashboard Force Rebuild", d) 

352 msg = "SYNC_ISE-ISE_FORCE_REBUILD" 

353 d.iseserver.force_rebuild = False 

354 d.iseserver.skip_update = True 

355 d.iseserver.save() 

356 if d.iseserver.last_sync and (d.iseserver.last_sync != d.iseserver.last_update): 

357 append_log(log, "ise_monitor::sync_ise::Database Config / Sync Timestamp Mismatch", d) 

358 msg = "SYNC_ISE-CONFIG_SYNC_TIMESTAMP_MISMATCH" 

359 if d.iseserver.last_sync and (d.iseserver.last_sync <= ctime): 

360 append_log(log, "ise_monitor::sync_ise::Past Manual Sync Interval", d) 

361 msg = "SYNC_ISE-PAST_SYNC_INTERVAL" 

362 

363 ingest_ise_data(dbs, log) 

364 

365 # After ingesting data, more updates may be required (Should these have been caught elsewhere?) 

366 sss = SyncSession.objects.all() 

367 for ss in sss: 

368 digest_database_data(ss, log) 

369 

370 append_log(log, "ise_monitor::sync_ise::Done") 

371 db_log("ise_monitor", log) 

372 return msg, log
